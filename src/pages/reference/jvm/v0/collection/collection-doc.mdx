# JVM - collection.doc()

Get a reference to a document in the collection.

```javascript
import { collection } from '@nitric/sdk'

const profiles = collection('profiles').for('reading', 'writing')

const drakesProfile = profiles.doc('Drake Mallard')
```

<CodeGroup>

```java
import io.nitric.Nitric;
import io.nitric.resources.CollectionPermission;

class User {
  String name;
  int age;

  public User(String name, int age) {
    this.name = name;
    this.age = age;
  }
}

public class Test {
  public static void main(String[] args) {
    var profiles = Nitric.INSTANCE.collection("profiles", User.class).with(CollectionPermission.Read, CollectionPermission.Write);

    var drakesProfile = profiles.doc("Drake Mallard");

    Nitric.INSTANCE.run();
  }
}
```

```kotlin
import io.nitric.Nitric
import io.nitric.resources.CollectionPermission

data class User(val name: String, val age: Int)

fun main() {
  val profiles = Nitric.collection("profiles", User::class.java).with(CollectionPermission.Read, CollectionPermission.Write)

  val drakesProfile = profiles.doc("Drake Mallard")

  Nitric.run()
}
```
