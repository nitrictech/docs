# Go - secret.version.access()

Retrieves the value from a version of a secret.

```go
import (
	"fmt"

	"github.com/nitrictech/go-sdk/resources"
)

func main() {
	secret, err := resources.NewSecret("secret-name").With(resources.SecretAccessing)
	if err != nil {
		fmt.Println(err)
		return
	}

	value, err := secret.Latest().Access(context.TODO())
	if err != nil {
		fmt.Println(err)
		return
	}

  fmt.Println(value.AsString())

	if err := resources.Run(); err != nil {
		fmt.Println(err)
	}
}
```

### Access a specific version of a secret

```go
import (
  "context"
	"fmt"

	"github.com/nitrictech/go-sdk/resources"
)

func main() {
	secret, err := resources.NewSecret("secret-name").With(resources.SecretAccessing)
	if err != nil {
		fmt.Println(err)
		return
	}

	version, err := secret.Version("the-version-id").Access(context.TODO())
  if err != nil {
    fmt.Println(err)
    return
  }

  fmt.Println(version.AsString())

	if err := resources.Run(); err != nil {
		fmt.Println(err)
	}
}
```
