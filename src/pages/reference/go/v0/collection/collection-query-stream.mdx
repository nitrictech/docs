# Go - collection.query.stream()

Process query results as a stream.

```go
import (
	"context"
	"fmt"

	"github.com/nitrictech/go-sdk/resources"
)

func main() {
	profiles, err := resources.NewCollection("profiles").With(resources.CollectionReading)
	if err != nil {
		return
	}

	stream, err := profiles.Query().Stream(context.TODO())
	if err != nil {
		return
	}

	if err := resources.Run(); err != nil {
		fmt.Println(err)
	}
}
```

## Examples

### Streaming results from a query

```go
import (
	"context"
	"fmt"

	"github.com/nitrictech/go-sdk/resources"
)

func main() {
	profiles, err := resources.NewCollection("profiles").With(resources.CollectionReading)
	if err != nil {
		return
	}

	stream, err := profiles.Query().Stream(context.TODO())
	if err != nil {
		return
	}

	for {
		res, err := stream.Next()
		if err != nil {
			fmt.Println("finished processing with: %w", err)
			break
		}

		fmt.Println(res.Content())
	}

	if err := resources.Run(); err != nil {
		fmt.Println(err)
	}
}
```
