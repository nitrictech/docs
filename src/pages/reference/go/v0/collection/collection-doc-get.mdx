# Go - collection.doc.get()

Retrieve the value of a document.

```go
import (
	"context"
	"fmt"

	"github.com/nitrictech/go-sdk/resources"
)

func main() {
	profiles, err := resources.NewCollection("profiles").With(resources.CollectionReading)
	if err != nil {
		return
	}

	drakeProfileRef := profiles.Doc("Drake Mallard")

	doc, err := drakeProfileRef.Get(context.TODO())
	if err != nil {
		return
	}

	fmt.Println(doc.Content())

	if err := resources.Run(); err != nil {
		fmt.Println(err)
	}
}
```
