# Go - bucket.file.delete()

Delete a file from a bucket.

```go
import (
	"context"
	"fmt"

	"github.com/nitrictech/go-sdk/resources"
)

func main() {
	bucket, err := resources.NewBucket("bucket-name").With(resources.BucketDeleting)
	if err != nil {
		return
	}

	err := bucket.File("cat.png").Delete(context.TODO())
	if err != nil {
		return
	}

	if err := resources.Run(); err != nil {
		fmt.Println(err)
	}
}
```
