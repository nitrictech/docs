# Go - bucket.file.getDownloadUrl()

Create a download url for a file within a bucket.

```go
import (
	"context"
	"fmt"

	"github.com/nitrictech/go-sdk/resources"
)

func main() {
	bucket, err := resources.NewBucket("bucket-name").With(resources.BucketReading)
	if err != nil {
		fmt.Println(err)
		return
	}

	uploadUrl, err := bucket.File("cat.png").DownloadUrl(context.TODO())
	if err != nil {
		fmt.Println(err)
		return
	}

	if err := resources.Run(); err != nil {
		fmt.Println(err)
	}
}
```

## Parameters

<Properties>
  <Property name="ctx" required type="context">
    The context of the call, used for tracing.
  </Property>
  <Property name="expiry" optional type="int">

### Redirect response to an image URL

```go
import (
	"context"
	"fmt"

	"github.com/nitrictech/go-sdk/faas"
	"github.com/nitrictech/go-sdk/resources"
)

func main() {
  api, err := resources.NewApi("main")
	if err != nil {
		fmt.Println(err)
		return
	}

	bucket, err := resources.NewBucket("images").With(resources.BucketReading)
	if err != nil {
		fmt.Println(err)
		return
	}

	api.Get("/images/:id", func(ctx *faas.HttpContext, _ faas.HttpHandler) (*faas.HttpContext, error) {
		id := ctx.Request.PathParams()["id"]

		downloadUrl, err := bucket.File(id).DownloadUrl(context.TODO())
		if err != nil {
			return ctx, err
		}

		ctx.Response.Headers["Location"] = []string{downloadUrl}
		ctx.Response.Status = 303

		return ctx, nil
	})

	if err := resources.Run(); err != nil {
		fmt.Println(err)
	}
}
```
